 def _parse_datetime(text: str) -> datetime:
     text = text.strip()
     formats = ["%d.%m.%Y %H:%M", "%d.%m %H:%M"]
     for fmt in formats:
         try:
             dt = datetime.strptime(text, fmt)
             if fmt == "%d.%m %H:%M":
                 dt = dt.replace(year=datetime.now().year)
             return dt
         except ValueError:
             continue
     raise ValueError("invalid format")
